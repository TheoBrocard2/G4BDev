// Méthode pour rafraîchir l'affichage de la liste des clients (exemple)
private void RefreshListeClients()
{
    // Supposons que lstClients soit un contrôle ListBox dans votre interface graphique
    // Vous pouvez mettre à jour son contenu avec la liste actuelle des clients enregistrés

    lstClients.Items.Clear(); // Effacer la liste actuelle

    // Récupérer la liste des clients enregistrés
    List<Client> clientsEnregistres = Client.GetClientsEnregistres();

    // Ajouter chaque client à la liste
    foreach (Client client in clientsEnregistres)
    {
        lstClients.Items.Add($"{client.Nom} {client.Prenom}");
    }
}
